213  // Constructs a tree out of the given n data points living in R^d.
214  explicit KmTree(int n, int d, Scalar *points): n_(n), d_(d), points_(points){
215  int node_size = sizeof(Node) + d_ * 3 * sizeof(Scalar);
216  node_data_ = (char*)malloc((2*n-1) * node_size);
217  point_indices_ = (int*)malloc(n * sizeof(int));
218  for (int i = 0; i < n; i++)
219  point_indices_[i] = i;
220  KM_ASSERT(node_data_ != 0 && point_indices_ != 0);
221 
222  // Calculate the bounding box for the points
223  Scalar *bound_v1 = PointAllocate(d_);
224  Scalar *bound_v2 = PointAllocate(d_);
225  KM_ASSERT(bound_v1 != 0 && bound_v2 != 0);
226  PointCopy(bound_v1, points, d_);
227  PointCopy(bound_v2, points, d_);
228  for (int i = 1; i < n; i++)
229  for (int j = 0; j < d; j++) {
230  if (bound_v1[j] > points[i*d_ + j]) bound_v1[j] = points[i*d_ + j];
231  if (bound_v2[j] < points[i*d_ + j]) bound_v1[j] = points[i*d_ + j];
232  }
233 
234  // Build the tree
235  char *temp_node_data = node_data_;
236  top_node_ = BuildNodes(points, 0, n-1, &temp_node_data);
237 
238  // Cleanup
239  PointFree(bound_v1);
240  PointFree(bound_v2);
241  }

245  free(node_data_);

248  // Given k cluster centers, this runs a full k-means iterations, choosing the next set of
249  // centers and returning the cost function for this set of centers. If assignment is not null,
250  // it should be an array of size n that will be filled with the index of the cluster (0 - k-1)
251  // that each data point is assigned to. The new center values will overwrite the old ones.
252  Scalar DoKMeansStep(int k, Scalar *centers, int *assignment) const {
253  Scalar *bad_center = PointAllocate(d_);
254  KM_ASSERT(bad_center != 0);
255  memset(bad_center, 0xff, d_ * sizeof(Scalar));
256 
257  // Allocate data
258  Scalar *sums = (Scalar*)calloc(k * d_, sizeof(Scalar));
259  int *counts = (int*)calloc(k, sizeof(int));
260  int num_candidates = 0;
261  int *candidates = (int*)malloc(k * sizeof(int));
262  KM_ASSERT(sums != 0 && counts != 0 && candidates != 0);
263  for (int i = 0; i < k; i++)
264  if (memcmp(centers + i*d_, bad_center, d_ * sizeof(Scalar)) != 0)
265  candidates[num_candidates++] = i;
266 
267  // Find nodes
268  Scalar result = DoKMeansStepAtNode(top_node_, num_candidates, candidates, centers, sums,
269  counts, assignment);
270 
271  // Set the new centers
272  for (int i = 0; i < k; i++) {
273  if (counts[i] > 0) {
274  PointScale(sums + i*d_, Scalar(1) / counts[i], d_);
275  PointCopy(centers + i*d_, sums + i*d_, d_);
276  } else {
277  memcpy(centers + i*d_, bad_center, d_ * sizeof(Scalar));
278  }
279  }
280 
281  // Cleanup memory
282  PointFree(bad_center);
283  free(candidates);
284  free(counts);
285  free(sums);
286  return result;
287  };

289  // Choose k initial centers for k-means using the kmeans++ seeding procedure. The resulting
290  // centers are returned via the centers variable, which should be pre-allocated to size k*d.
291  // The cost of the initial clustering is returned.
292  Scalar SeedKMeansPlusPlus(int k, Scalar *centers) const{
293  Scalar *dist_sq = (Scalar*)malloc(n_ * sizeof(Scalar));
294  KM_ASSERT(dist_sq != 0);
295 
296  // Choose an initial center uniformly at random
297  SeedKmppSetClusterIndex(top_node_, 0);
298  int i = GetRandom(n_);
299  memcpy(centers, points_ + point_indices_[i]*d_, d_*sizeof(Scalar));
300  Scalar total_cost = 0;
301  for (int j = 0; j < n_; j++) {
302  dist_sq[j] = PointDistSq(points_ + point_indices_[j]*d_, centers, d_);
303  total_cost += dist_sq[j];
304  }
305 
306  // Repeatedly choose more centers
307  for (int new_cluster = 1; new_cluster < k; new_cluster++) {
308  while (1) {
309  Scalar cutoff = (rand() / Scalar(RAND_MAX)) * total_cost;
310  Scalar cur_cost = 0;
311  for (i = 0; i < n_; i++) {
312  cur_cost += dist_sq[i];
313  if (cur_cost >= cutoff)
314  break;
315  }
316  if (i < n_)
317  break;
318  }
319  memcpy(centers + new_cluster*d_, points_ + point_indices_[i]*d_, d_*sizeof(Scalar));
320  total_cost = SeedKmppUpdateAssignment(top_node_, new_cluster, centers, dist_sq);
321  }
322 
323  // Clean up and return
324  free(dist_sq);
325  return total_cost;
326  };

339  // Helper functions for constructor
340  Node *BuildNodes(Scalar *points, int first_index, int last_index, char **next_node_data) {
341  // Allocate the node
342  Node *node = (Node*)(*next_node_data);
343  (*next_node_data) += sizeof(Node);
344  node->sum = (Scalar*)(*next_node_data);
345  (*next_node_data) += sizeof(Scalar) * d_;
346  node->median = (Scalar*)(*next_node_data);
347  (*next_node_data) += sizeof(Scalar) * d_;
348  node->radius = (Scalar*)(*next_node_data);
349  (*next_node_data) += sizeof(Scalar) * d_;
350 
351  // Fill in basic info
352  node->num_points = (last_index - first_index + 1);
353  node->first_point_index = first_index;
354 
355  // Calculate the bounding box
356  Scalar *first_point = points + point_indices_[first_index] * d_;
357  Scalar *bound_p1 = PointAllocate(d_);
358  Scalar *bound_p2 = PointAllocate(d_);
359  KM_ASSERT(bound_p1 != 0 && bound_p2 != 0);
360  PointCopy(bound_p1, first_point, d_);
361  PointCopy(bound_p2, first_point, d_);
362  for (int i = first_index+1; i <= last_index; i++)
363  for (int j = 0; j < d_; j++) {
364  Scalar c = points[point_indices_[i]*d_ + j];
365  if (bound_p1[j] > c) bound_p1[j] = c;
366  if (bound_p2[j] < c) bound_p2[j] = c;
367  }
368 
369  // Calculate bounding box stats and delete the bounding box memory
370  Scalar max_radius = -1;
371  int split_d = -1;
372  for (int j = 0; j < d_; j++) {
373  node->median[j] = (bound_p1[j] + bound_p2[j]) / 2;
374  node->radius[j] = (bound_p2[j] - bound_p1[j]) / 2;
375  if (node->radius[j] > max_radius) {
376  max_radius = node->radius[j];
377  split_d = j;
378  }
379  }
380  PointFree(bound_p2);
381  PointFree(bound_p1);
382 
383  // If the max spread is 0, make this a leaf node
384  if (max_radius == 0) {
385  node->lower_node = node->upper_node = 0;
386  PointCopy(node->sum, first_point, d_);
387  if (last_index != first_index)
388  PointScale(node->sum, Scalar(last_index - first_index + 1), d_);
389  node->opt_cost = 0;
390  return node;
391  }
392 
393  // Partition the points around the midpoint in this dimension. The partitioning is done in-place
394  // by iterating from left-to-right and right-to-left in the same way that partioning is done for
395  // quicksort.
396  Scalar split_pos = node->median[split_d];
397  int i1 = first_index, i2 = last_index, size1 = 0;
398  while (i1 <= i2) {
399  bool is_i1_good = (points[point_indices_[i1]*d_ + split_d] < split_pos);
400  bool is_i2_good = (points[point_indices_[i2]*d_ + split_d] >= split_pos);
401  if (!is_i1_good && !is_i2_good) {
402  int temp = point_indices_[i1];
403  point_indices_[i1] = point_indices_[i2];
404  point_indices_[i2] = temp;
405  is_i1_good = is_i2_good = true;
406  }
407  if (is_i1_good) {
408  i1++;
409  size1++;
410  }
411  if (is_i2_good) {
412  i2--;
413  }
414  }
415 
416  // Create the child nodes
417  KM_ASSERT(size1 >= 1 && size1 <= last_index - first_index);
418  node->lower_node = BuildNodes(points, first_index, first_index + size1 - 1, next_node_data);
419  node->upper_node = BuildNodes(points, first_index + size1, last_index, next_node_data);
420 
421  // Calculate the new sum and opt cost
422  PointCopy(node->sum, node->lower_node->sum, d_);
423  PointAdd(node->sum, node->upper_node->sum, d_);
424  Scalar *center = PointAllocate(d_);
425  KM_ASSERT(center != 0);
426  PointCopy(center, node->sum, d_);
427  PointScale(center, Scalar(1) / node->num_points, d_);
428  node->opt_cost = GetNodeCost(node->lower_node, center) + GetNodeCost(node->upper_node, center);
429  PointFree(center);
430  return node;
431  };
432  Scalar GetNodeCost(const Node *node, Scalar *center) const {
433  Scalar dist_sq = 0;
434  for (int i = 0; i < d_; i++) {
435  Scalar x = (node->sum[i] / node->num_points) - center[i];
436  dist_sq += x*x;
437  }
438  return node->opt_cost + node->num_points * dist_sq;
439  };

441  // Helper functions for DoKMeans step
442  Scalar DoKMeansStepAtNode(const Node *node, int k, int *candidates, Scalar *centers,
443  Scalar *sums, int *counts, int *assignment) const {
444  // Determine which center the node center is closest to
445  Scalar min_dist_sq = PointDistSq(node->median, centers + candidates[0]*d_, d_);
446  int closest_i = candidates[0];
447  for (int i = 1; i < k; i++) {
448  Scalar dist_sq = PointDistSq(node->median, centers + candidates[i]*d_, d_);
449  if (dist_sq < min_dist_sq) {
450  min_dist_sq = dist_sq;
451  closest_i = candidates[i];
452  }
453  }
454 
455  // If this is a non-leaf node, recurse if necessary
456  if (node->lower_node != 0) {
457  // Build the new list of candidates
458  int new_k = 0;
459  int *new_candidates = (int*)malloc(k * sizeof(int));
460  KM_ASSERT(new_candidates != 0);
461  for (int i = 0; i < k; i++)
462  if (!ShouldBePruned(node->median, node->radius, centers, closest_i, candidates[i]))
463  new_candidates[new_k++] = candidates[i];
464 
465  // Recurse if there's at least two
466  if (new_k > 1) {
467  Scalar result = DoKMeansStepAtNode(node->lower_node, new_k, new_candidates, centers,
468  sums, counts, assignment) +
469  DoKMeansStepAtNode(node->upper_node, new_k, new_candidates, centers,
470  sums, counts, assignment);
471  free(new_candidates);
472  return result;
473  } else {
474  free(new_candidates);
475  }
476  }
477 
478  // Assigns all points within this node to a single center
479  PointAdd(sums + closest_i*d_, node->sum, d_);
480  counts[closest_i] += node->num_points;
481  if (assignment != 0) {
482  for (int i = node->first_point_index; i < node->first_point_index + node->num_points; i++)
483  assignment[point_indices_[i]] = closest_i;
484  }
485  return GetNodeCost(node, centers + closest_i*d_);
486  };
487  bool ShouldBePruned(Scalar *box_median, Scalar *box_radius, Scalar *centers, int best_index,
488  int test_index) const {
489  if (best_index == test_index)
490  return false;
491 
492  Scalar *best = centers + best_index*d_;
493  Scalar *test = centers + test_index*d_;
494  Scalar lhs = 0, rhs = 0;
495  for (int i = 0; i < d_; i++) {
496  Scalar component = test[i] - best[i];
497  lhs += component * component;
498  if (component > 0)
499  rhs += (box_median[i] + box_radius[i] - best[i]) * component;
500  else
501  rhs += (box_median[i] - box_radius[i] - best[i]) * component;
502  }
503  return (lhs >= 2*rhs);
504  };

514  Scalar SeedKmppUpdateAssignment(const Node *node, int new_cluster, Scalar *centers,
515  Scalar *dist_sq) const {
516  // See if we can assign all points in this node to one cluster
517  if (node->kmpp_cluster_index >= 0) {
518  if (ShouldBePruned(node->median, node->radius, centers, node->kmpp_cluster_index, new_cluster))
519  return GetNodeCost(node, centers + node->kmpp_cluster_index*d_);
520  if (ShouldBePruned(node->median, node->radius, centers, new_cluster,
521  node->kmpp_cluster_index)) {
522  SeedKmppSetClusterIndex(node, new_cluster);
523  for (int i = node->first_point_index; i < node->first_point_index + node->num_points; i++)
524  dist_sq[i] = PointDistSq(points_ + point_indices_[i]*d_, centers + new_cluster*d_, d_);
525  return GetNodeCost(node, centers + new_cluster*d_);
526  }
527 
528  // It may be that the a leaf-node point is equidistant from the new center or old
529  if (node->lower_node == 0)
530  return GetNodeCost(node, centers + node->kmpp_cluster_index*d_);
531  }
532 
533  // Recurse
534  Scalar cost = SeedKmppUpdateAssignment(node->lower_node, new_cluster, centers, dist_sq) +
535  SeedKmppUpdateAssignment(node->upper_node, new_cluster, centers, dist_sq);
536  int i1 = node->lower_node->kmpp_cluster_index, i2 = node->upper_node->kmpp_cluster_index;
537  if (i1 == i2 && i1 != -1)
538  node->kmpp_cluster_index = i1;
539  else
540  node->kmpp_cluster_index = -1;
541  return cost;};

556  // See KMeans.h
557  // Performs one full execution of k-means, logging any relevant information, and tracking meta
558  // statistics for the run. If min or max values are negative, they are treated as unset.
559  // best_centers and best_assignment can be 0, in which case they are not set.
560  inline void RunKMeansOnce(const KmTree &tree, int n, int k, int d, Scalar *points, Scalar *centers,
561  Scalar *min_cost, Scalar *max_cost, Scalar *total_cost,
562  double start_time, double *min_time, double *max_time,
563  double *total_time, Scalar *best_centers, int *best_assignment) {
564  const Scalar kEpsilon = Scalar(1e-8); // Used to determine when to terminate k-means
565 
566  // Do iterations of k-means until the cost stabilizes
567  Scalar old_cost = 0;
568  bool is_done = false;
569  for (int iteration = 0; !is_done; iteration++) {
570  Scalar new_cost = tree.DoKMeansStep(k, centers, 0);
571  is_done = (iteration > 0 && new_cost >= (1 - kEpsilon) * old_cost);
572  old_cost = new_cost;
573  }
574  double this_time = GetSeconds() - start_time;
575 
576  // Log the clustering we found
577 
578  // Handle a new min cost, updating best_centers and best_assignment as appropriate
579  if (*min_cost < 0 || old_cost < *min_cost) {
580  *min_cost = old_cost;
581  if (best_assignment != 0)
582  tree.DoKMeansStep(k, centers, best_assignment);
583  if (best_centers != 0)
584  memcpy(best_centers, centers, sizeof(Scalar)*k*d);
585  }
586 
587  // Update all other aggregate stats
588  if (*max_cost < old_cost) *max_cost = old_cost;
589  *total_cost += old_cost;
590  if (*min_time < 0 || *min_time > this_time)
591  *min_time = this_time;
592  if (*max_time < this_time) *max_time = this_time;
593  *total_time += this_time;
594  }

597  // Runs k-means on the given set of points.
598  // - n: The number of points in the data set
599  // - k: The number of clusters to look for
600  // - d: The number of dimensions that the data set lives in
601  // - points: An array of size n*d where points[d*i + j] gives coordinate j of point i
602  // - attempts: The number of times to independently run k-means with different starting centers.
603  // The best result is always returned (as measured by the cost function).
604  // - centers: This can either be null or an array of size k*d. In the latter case, it will be
605  // filled with the locations of all final cluster centers. Specifically
606  // centers[d*i + j] will give coordinate j of center i. If the cluster is unused, it
607  // will contain NaN instead.
608  // - assignments: This can either be null or an array of size n. In the latter case, it will be
609  // filled with the cluster that each point is assigned to (an integer between 0
610  // and k-1 inclusive).
611  // The final cost of the clustering is also returned.
612  // See KMeans.h
613  Scalar RunKMeans(int n, int k, int d, Scalar *points, int attempts,
614  Scalar *ret_centers, int *ret_assignment) {
615  KM_ASSERT(k >= 1);
616 
617  // Create the tree and log
618  KmTree tree(n, d, points);
619 
620  // Initialization
621  Scalar *centers = (Scalar*)malloc(sizeof(Scalar)*k*d);
622  int *unused_centers = (int*)malloc(sizeof(int)*n);
623  KM_ASSERT(centers != 0 && unused_centers != 0);
624  Scalar min_cost = -1, max_cost = -1, total_cost = 0;
625  double min_time = -1, max_time = -1, total_time = 0;
626 
627  // Handle k > n
628  if (k > n) {
629  memset(centers + n*d, -1, (k-d)*sizeof(Scalar));
630  k = n;
631  }
632 
633  // Run all the attempts
634  for (int attempt = 0; attempt < attempts; attempt++) {
635  double start_time = GetSeconds();
636 
637  // Choose centers uniformly at random
638  for (int i = 0; i < n; i++)
639  unused_centers[i] = i;
640  int num_unused_centers = n;
641  for (int i = 0; i < k; i++) {
642  int j = GetRandom(num_unused_centers--);
643  memcpy(centers + i*d, points + unused_centers[j]*d, d*sizeof(Scalar));
644  unused_centers[j] = unused_centers[num_unused_centers];
645  }
646 
647  // Run k-means
648  RunKMeansOnce(tree, n, k, d, points, centers, &min_cost, &max_cost, &total_cost, start_time,
649  &min_time, &max_time, &total_time, ret_centers, ret_assignment);
650  }
651  // Clean up and return
652  free(unused_centers);
653  free(centers);
654  return min_cost;
655  }

746  template <class LIST_OF_VECTORS1,class LIST_OF_VECTORS2>
747  double stub_kmeans(
748  const bool use_kmeansplusplus_method,
749  const size_t k,

766  const typename LIST_OF_VECTORS1::const_iterator it_end =points.end();

768  typedef typename LIST_OF_VECTORS2::value_type TInnerVectorCenters;

780  }

784  }

787  trg_ptr+=dims;

791  const double ret = detail::internal_kmeans(false,N,k,points.begin()->size(),&raw_vals[0],attempts,&centers[0],&assignments[0]);
